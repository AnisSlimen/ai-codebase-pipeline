// Example contract test pattern using RestAssured + JSON Schema validator
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@Testcontainers
class IngestSymbolsContractTest {
@Test
void postIngestSymbols_respondsWithJobIdAndProjectId() {
// Load the fixture symbols.json
String symbolsJson = loadFixture('/test-fixtures/small-project/expected-symbols.json');
given()
.header('X-API-Key', DEV_API_KEY)
.contentType(ContentType.JSON)
.body(symbolsJson)
.when()
.post('/api/v1/ingest/symbols')
.then()
.statusCode(200)
// Validate response shape against OpenAPI schema
.body(matchesJsonSchema(openApiResponseSchema('PostIngestSymbols', 200)));
}
}
