OUTPUT 3: /contracts/db-schema.sql
-- AI Codebase Architect Plugin — Canonical DB Schema v1.0
-- AGENT-INFRA implements this as Flyway V1__init.sql exactly as written here.
CREATE TABLE users (
id            BIGSERIAL PRIMARY KEY,
email         TEXT NOT NULL UNIQUE,
api_key_hash  TEXT NOT NULL,
plan_tier     TEXT NOT NULL DEFAULT 'FREE',
query_count   INT NOT NULL DEFAULT 0,
created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE TABLE api_keys (
id           BIGSERIAL PRIMARY KEY,
user_id      BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
key_prefix   TEXT NOT NULL,
key_hash     TEXT NOT NULL UNIQUE,
last_used_at TIMESTAMPTZ,
active       BOOLEAN NOT NULL DEFAULT TRUE,
created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE TABLE projects (
id               BIGSERIAL PRIMARY KEY,
name             TEXT NOT NULL,
user_id          BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
language         TEXT NOT NULL DEFAULT 'JAVA',
status           TEXT NOT NULL DEFAULT 'PENDING',
created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
last_analyzed_at TIMESTAMPTZ
);
CREATE TABLE analysis_jobs (
id           BIGSERIAL PRIMARY KEY,
project_id   BIGINT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
status       TEXT NOT NULL DEFAULT 'PENDING',
progress_pct INT NOT NULL DEFAULT 0,
error_msg    TEXT,
started_at   TIMESTAMPTZ,
completed_at TIMESTAMPTZ
);
CREATE TABLE symbols (
id           BIGSERIAL PRIMARY KEY,
project_id   BIGINT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
file_path    TEXT NOT NULL,
class_name   TEXT NOT NULL,
package_name TEXT NOT NULL,
layer_type   TEXT NOT NULL,
annotations  TEXT[] NOT NULL DEFAULT '{}',
start_line   INT
);
CREATE INDEX idx_symbols_project ON symbols(project_id);
CREATE INDEX idx_symbols_layer   ON symbols(project_id, layer_type);
CREATE TABLE symbol_methods (
id          BIGSERIAL PRIMARY KEY,
symbol_id   BIGINT NOT NULL REFERENCES symbols(id) ON DELETE CASCADE,
name        TEXT NOT NULL,
return_type TEXT,
params      TEXT[] NOT NULL DEFAULT '{}'
);
CREATE TABLE symbol_fields (
id        BIGSERIAL PRIMARY KEY,
symbol_id BIGINT NOT NULL REFERENCES symbols(id) ON DELETE CASCADE,
name      TEXT NOT NULL,
type      TEXT NOT NULL
);
CREATE TABLE dependencies (
id               BIGSERIAL PRIMARY KEY,
project_id       BIGINT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
source_symbol_id BIGINT NOT NULL REFERENCES symbols(id) ON DELETE CASCADE,
target_symbol_id BIGINT NOT NULL REFERENCES symbols(id) ON DELETE CASCADE,
edge_type        TEXT NOT NULL,
edge_count       INT NOT NULL DEFAULT 1
);
CREATE INDEX idx_deps_project ON dependencies(project_id);
CREATE INDEX idx_deps_source  ON dependencies(source_symbol_id);
CREATE TABLE violations (
id               BIGSERIAL PRIMARY KEY,
project_id       BIGINT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
rule_id          TEXT NOT NULL,
source_symbol_id BIGINT REFERENCES symbols(id) ON DELETE CASCADE,
target_symbol_id BIGINT REFERENCES symbols(id) ON DELETE CASCADE,
severity         TEXT NOT NULL,
suppressed       BOOLEAN NOT NULL DEFAULT FALSE,
detected_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE TABLE query_logs (
id           BIGSERIAL PRIMARY KEY,
project_id   BIGINT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
user_id      BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
question     TEXT NOT NULL,
answer       TEXT,
tokens_used  INT,
created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
